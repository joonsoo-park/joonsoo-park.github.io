- # 이미지 로드
- ```bash
  docker load -i exemone-sparselog-v3.0.604.0.tar.gz
  ```
-
- # Dockerfile 추가
- `${EXEMONE_HOME}/dockerfile/exemone-sparselog/Dockerfile`
- ```bash
  mkdir ${EXEMONE_HOME}/dockerfile/exemone-sparselog
  
  cat > Dockerfile <<'EOF'
  ARG SPARSELOG_IMAGE
  ARG SPARSELOG_VERSION
  FROM ${SPARSELOG_IMAGE}:${SPARSELOG_VERSION}
  
  ARG EXEMONE_USER_UID
  ARG EXEMONE_USER_GID
  
  USER root
  
  # 설치 환경에 따른 uid, gid 변경
  RUN usermod -u ${EXEMONE_USER_UID} exemone && \
      groupmod -g ${EXEMONE_USER_GID} exemone
  
  RUN chown -R exemone:exemone /exemone-sparselog
  
  USER exemone
  EOF
  ```
-
- # .env 파일 sparselog 추가
- ```env
  SPARSELOG_IMAGE=maxgauge/exemone-sparselog
